       8 /**
       9  * Обработчик событий TSR (Traffic Sign Recognition)
      10  *
      11  * Распознавание дорожных знаков через камеру автомобиля.
      12  * Данные поступают через CAN-bus в виде событий.
      13  *
      14  * @param canBusManager Менеджер CanBusService для регистрации callback
      15  * @param callback Callback для получения событий (может быть null)
      16  */
      17 class TSREventHandler(
      18     private val canBusManager: CanBusServiceManager,
      19     private val callback: TSRCallback? = null
      20 ) {
      21
      22     // Текущее состояние
      23     private var currentSpeedLimit = 0
      24     private var currentSignType = 0
      25     private var tsrOperatingStatus = 0
      26     private var isTsrAvailable = false
      27
      28     private val listener = object : CanBusListener() {
      29
      30         override fun onVehicleStateChanged(vehicle: VehicleState, IState: Int) {
      31             handleVehicleStateChanged(vehicle, IState)
      32         }
      33
      34         override fun onVehicleSpeedChanged(speed: Int) {
      35             // Можно использовать для сравжения с лимитом
      36             Log.d(TAG, "🚗 Скорость авто: $speed км/ч")
      37         }
      38     }
      39
      40     private fun handleVehicleStateChanged(vehicle: VehicleState, IState: Int) {
      41         when (vehicle) {
      42             // 🚸 Ограничение скорости с камеры
      43             VehicleState.TSR_SPEED_LIMIT -> {
      44                 currentSpeedLimit = IState
      45                 Log.i(TAG, "🚸 TSR_SPEED_LIMIT: $currentSpeedLimit км/ч")
      46                 callback?.onSpeedLimitChanged(currentSpeedLimit)
      47             }
      48
      49             // 🚦 Тип распознанного знака
      50             VehicleState.TSR_SIGN_TYPE -> {
      51                 currentSignType = IState
      52                 Log.i(TAG, "🚦 TSR_SIGN_TYPE: $currentSignType (${getSignTypeName(currentSignType)})")
      53                 callback?.onSignTypeChanged(currentSignType)
      54             }
      55
      56             // 📊 Единицы измерения
      57             VehicleState.TSR_SPEED_LIMIT_UNIT -> {
      58                 val unit = if (IState == 0) "км/ч" else "mph"
      59                 Log.d(TAG, "📊 TSR_SPEED_LIMIT_UNIT: $unit")
      60             }
      61
      62             // ⚠️ Предупреждение TSR
      63             VehicleState.TSR_WARNING -> {
      64                 if (IState == 1) {
      65                     Log.w(TAG, "⚠️ TSR_WARNING: Предупреждение!)
      66                     callback?.onTsrWarning()
      67                 }
      68             }
      69
      70             // 🔧 Статус работы системы
      71             VehicleState.TSR_OPERATING_STATUS -> {
      72                 tsrOperatingStatus = IState
      73                 isTsrAvailable = (IState != DFVehicleState.TSR_OPERATING_STATUS_FAILURE &&
      74                                   IState != DFVehicleState.TSR_OPERATING_CAREMA_BLOCKED &&
      75                                   IState != DFVehicleState.TSR_OPERATING_PERMANENT_FAILURE)
      76
      77                 Log.d(TAG, "🔧 TSR_OPERATING_STATUS: $IState (${getOperatingStatusName(IState)})")
      78                 callback?.onOperatingStatusChanged(IState, isTsrAvailable)
      79             }
      80
      81             // 📡 Общий статус
      82             VehicleState.TSRStatus -> {
      83                 Log.d(TAG, "📡 TSRStatus: $IState")
      84             }
      85
      86             // 🔔 Тип предупреждения
      87             VehicleState.TSR_WARNING_TYPE -> {
      88                 Log.d(TAG, "🔔 TSR_WARNING_TYPE: $IState")
      89             }
      90
      91             else -> {}
      92         }
      93     }
      94
      95     // === Вспомогательные методы ===
      96
      97     /**
      98      * Получить название статуса работы TSR
      99      */
     100     fun getOperatingStatusName(status: Int): String {
     101         return when (status) {
     102             DFVehicleState.TSR_OPERATING_STATUS_OFF -> "OFF"
     103             DFVehicleState.TSR_OPERATING_STATUS_FUSION_MODE -> "FUSION_MODE (камера+навигация)"
     104             DFVehicleState.TSR_OPERATING_STATUS_VISION_ONLY_MODE -> "VISION_ONLY (только камера)"
     105             DFVehicleState.TSR_OPERATING_STATUS_NAVIGATION_ONLY_MODE -> "NAVIGATION_ONLY (только навигация)"
     106             DFVehicleState.TSR_OPERATING_STATUS_FAILURE -> "FAILURE (ошибка)"
     107             DFVehicleState.TSR_OPERATING_CAREMA_BLOCKED -> "CAMERA_BLOCKED (заблокирована)"
     108             DFVehicleState.TSR_OPERATING_PERMANENT_FAILURE -> "PERMANENT_FAILURE (неустранимая ошибка)"
     109             DFVehicleState.TSR_OPERATING_NOT_CONFIG -> "NOT_CONFIG (не настроено)"
     110             else -> "UNKNOWN ($status)"
     111         }
     112     }
     113
     114     /**
     115      * Получить название типа знака
     116      */
     117     fun getSignTypeName(signType: Int): String {
     118         return when (signType) {
     119             0 -> "Нет знака"
     120             1 -> "Ограничение скорости"
     121             2 -> "Конец зоны ограничения"
     122             3 -> "Запрет обгона"
     123             4 -> "Конец запрета обгона"
     124             5 -> "STOP"
     125             6 -> "Уступи дорогу"
     126             7 -> "Другой знак"
     127             else -> "Неизвестно ($signType)"
     128         }
     129     }
     130
     131     // === Регистрация / Отписка ===
     132
     133     private var isRegistered = false
     134
     135     /**
     136      * Зарегистрировать callback в CanBusService
     137      * @return true если успешно
     138      */
     139     fun register(): Boolean {
     140         if (isRegistered) return true
     141         val success = canBusManager.registerCallback(listener)
     142         if (success) {
     143             isRegistered = true
     144             Log.i(TAG, "✅ TSREventHandler registered")
     145         }
     146         return success
     147     }
     148
     149     /**
     150      * Отписаться от callback
     151      * @return true если успешно
     152      */
     153     fun unregister(): Boolean {
     154         if (!isRegistered) return true
     155         val success = canBusManager.unregisterCallback(listener)
     156         if (success) {
     157             isRegistered = false
     158             Log.i(TAG, "❌ TSREventHandler unregistered")
     159         }
     160         return success
     161     }
     162
     163     // === Геттеры текущего состояния ===
     164
     165     /**
     166      * Получить текущий лимит скорости (последний распознанный знак)
     167      * @return ограничение в км/ч, 0 если знак не распознан
     168      */

     171     /**

     177     /**

     183     /**

     186      */

	# Анализ архитектуры TTS-сервиса в `com/qinggan/ttsservice`

## Обзор

Система поддерживает **множество TTS-движков одновременно**, с возможностью переключения в рантайме.  
Используется шаблон **Strategy + Factory**, где:
- Все движки реализуют общий интерфейс через `AbstractTtsImpl`
- `ImplFactory` выбирает активный движок на основе настроек
- `TtsService` — Android-сервис, управляющий жизненным циклом синтеза

---

## Ключевые компоненты

### 1. `TtsService.java`
- **Тип**: `android.app.Service`
- **Назначение**: Центральный TTS-сервис, обрабатывающий запросы
- **Компоненты**:
  - `RequestQueue` — очередь запросов синтеза
  - `TtsImplPool` — пул реализаций движков ( Singleton / pool )
  - `media player manager` — управляет воспроизведением

### 2. `ImplFactory.java`
- **Тип**: фабрика движков
- **Роль**:
  - Выбирает TTS-движок по текущим настройкам (хранятся в SQLite)
  - Поддерживает **гибридные реализации**:
    - `NuanceXiaoIceImpl`
    - `AISpeechXiaoIceImpl`
    - `ArkLitePateoTtsImpl`
    - и др.

### 3. Реализации движков (`impl/`)
| Папка | Движок | Особенности |
|-------|--------|-------------|
| `impl/baidu/` | Baidu TTS | `BaiduTtsImpl.java`, `BaiduUtil.java` |
| `impl/iflytek/` | iFLYTEK (科大讯飞) | Native-биндинг (`IflytekNativeImp.java`), `ITTSPlayListener` |
| `impl/microsoft/` | MS Speech SDK | Полная поддержка SDK v1.x, обработчики ошибок |
| `impl/aispeech/` | AiSpeech | `AISpeechTtsImpl`, аутентификация (`AISpeechAuth`) |
| `impl/arklite/` | ArkLite | Собственный движок (Pateo), PCM-настройки, плейер |
| `impl/xiaoice/` | XiaoIce | Microsoft-совместимый, но кастомная реализация |
| `impl/pateo/` | Pateo | Родной движок для Geely/Pateo |
| `impl/localaudio/` | Локальные файлы | Не требует интернета, `AudioTrack` / `MediaPlayer` |
| `impl/combine/` | Гибриды | Например, Nuance → распознавание, XiaoIce → синтез |

### 4. Кэширование
- `impl/cache/CacheTtsImpl.java` — кэширует результаты синтеза
- Метаданные в `sql/TtsImplSetting.java`

### 5. Отладка и дампы
- `sde/TtsSdeMgr.java`, `sde/TtsDebugReceiver.java`
- `util/QGTtsPcmDump.java` — запись PCM-данных в `/sdcard/` (включается/выключается)

### 6. Настройки и UI
- `sql/TtsSettingsHelper.java`, `TtsContentProvider.java`, `TtsObserverSettingsHelper.java`
- `ui/EngineSettings.java` — экран переключения движка (возможно, в настройках приложения)

### 7. Вспомогательные утилиты
- `util/`:
  - `Config.java` — глобальные настройки
  - `DeviceIdUtils.java` — ID устройства
  - `HttpDownloader.java` — загрузка ресурсов (например, голосов)
  - `TtsFixHelper.java` — хаки для багов на определённых устройствах
- `platform/PlatformAdaptMgr.java` — адаптация под платформу ( Automotive / Android )
